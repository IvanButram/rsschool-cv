def count_positives_sum_negatives(arr):
    coun=0
    sum=0
    for i in arr:
        if i > 0:
            coun=coun+1
        elif i < 0:
            sum=sum+i
    if arr==[] and sum==0 and coun==0:
        arr=[]
    else:
        arr=[coun,sum]
    return arr
